import React from 'react';
import {useAuth0} from '@auth0/auth0-react';


const Profile = ()=> {
	const {user, isAuthenticated} = useAuth0();
	return(
		isAuthenticated && ( 
		<>
		 	<img src = {user.picture} />
		 	<h3> {user.name} </h3>
		 	<p> {user.email} </p>
			{JSON.stringify(user, null, 2 )}

		</>
		)
		)
}

export default Profile;
